// ========================================== // LÓGICA DE LOGIN COM VERIFICAÇÃO DE BLOQUEIO // ========================================== async function fazerLogin() { const inputWhatsapp = document.getElementById('campo-whatsapp').value.replace(/\D/g, ''); const msgErro = document.getElementById('msg-erro-login'); // Crie uma div com esse id para mostrar erros if (inputWhatsapp.length < 10) { msgErro.innerText = "Digite um WhatsApp válido."; msgErro.style.display = "block"; return; } msgErro.style.display = "none"; msgErro.innerText = "Entrando..."; const emailFake = inputWhatsapp + "@app.ndvi"; // A senha padrão que você cadastrou para eles (ex: 123456 ou o cpf sem pontuação) const senha = inputWhatsapp; // Altere aqui se a senha for diferente do whatsapp try { // 1. Tenta logar no Supabase Auth const { data: authData, error: authError } = await supabase.auth.signInWithPassword({ email: emailFake, password: senha }); if (authError) throw authError; // 2. Se logou, busca os dados na tabela 'clientes' para ver se está BLOQUEADO const { data: cliente, error: dbError } = await supabase .from('clientes') .select('id, ativo') .eq('whatsapp', inputWhatsapp) .single(); if (dbError || !cliente) { await supabase.auth.signOut(); throw new Error("Usuário não encontrado na base de dados."); } // 3. VERIFICA SE ESTÁ BLOQUEADO (PLANO VENCIDO) if (cliente.ativo === false) { await supabase.auth.signOut(); // Desloga imediatamente msgErro.innerText = "⚠️ SEU ACESSO ESTÁ BLOQUEADO. Plano expirado. Entre em contato com o suporte."; msgErro.style.display = "block"; return; } // 4. Se chegou aqui, está tudo ok! Redireciona ou carrega o mapa msgErro.style.display = "none"; carregarTelaDoMapa(cliente.id); // Chame a sua função que mostra a lista de talhões } catch (error) { msgErro.innerText = "Erro ao entrar: " + error.message; msgErro.style.display = "block"; } }
